4. **Test script structure**:
```python
#!/usr/bin/env python3
"""Test FileSummarizer metadata generation on real files."""

import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent.parent.parent / "auzoom" / "src"))

from auzoom.mcp.file_summarizer import FileSummarizer

# Test cases
test_files = [
    ("README.md", "markdown"),
    ("VALIDATION-SUMMARY.md", "markdown"),
    (".planning/config.json", "json"),
    ("pyproject.toml", "toml"),
]

def test_file_summarizer():
    """Test metadata generation on real files."""
    summarizer = FileSummarizer(cache_dir=Path("/tmp/auzoom-test-cache"))

    results = []
    for file_path, file_type in test_files:
        path = Path(file_path)
        if not path.exists():
            continue

        content = path.read_text()
        lines = content.splitlines()

        # Generate metadata
        metadata = summarizer._generate_summary_text(path, lines, content)

        # Calculate metrics
        full_size = len(content)
        metadata_size = len(metadata)
        reduction_pct = (1 - metadata_size / full_size) * 100

        results.append({
            "file": file_path,
            "type": file_type,
            "full_size": full_size,
            "metadata_size": metadata_size,
            "reduction": f"{reduction_pct:.1f}%",
            "metadata": metadata
        })

    return results

if __name__ == "__main__":
    results = test_file_summarizer()

    # Print results
    for r in results:
        print(f"\n=== {r['file']} ===")
        print(f"Type: {r['type']}")
        print(f"Full size: {r['full_size']} bytes")
        print(f"Metadata size: {r['metadata_size']} bytes")
        print(f"Reduction: {r['reduction']}")
        print(f"\nMetadata:\n{r['metadata']}")
```
